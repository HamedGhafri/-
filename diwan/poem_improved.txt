دين الحياة
@روحانية
إذا المـــرء قـــد كَـــلَّ ديــن الحـــيــاةْ
ويــأمــل فيها بــأن لا يــذمــــــــــــــــــــــــــْ

فــحــبل الإلــه طـــويــلٌ مـــــداهــــــــــْ
مـــتـى مــا يـــشــاء فـــمــنــه الــكــرمــْ

وإنــي لــمُــفــهــم مــن فــي عــمــاهــــــــــْ
ومُـــســـمـــع مـــن فـــي الــبــرايــا الأصــمــْ

فـــهـــذي وصــايــا تــجــرُّ الـــهـــداةـــــــــْ
وتــظــمــؤ مــن قـــد عــصــاه الــحــمــمــْ

إذا كـــان لــلــمــرء وجـــهٌ يــــراهــــــــــْ
فـــوجــه الإلـــه يــنــيــر الــظُــلَـــمــــــــْ

وإن كــان عـــســـرٌ إلـــيـــه اجــتــبــاهْ
فــدرب الإلـــه يـــســيــر الــــــرزمــــــــْ

ومـــا كــل عـــبــد لــديــه ســُـــــــــواةْ
وتـــقـــواه فــي ذا تـــكـــون الـــحـــكـــمْ

وعـــظـــم قـــلــوبــا لـــديــه ثـــقــاهـــــــْ
كـــأمٍّ تـــكـــون عــلــيــنــا نــعـــمــــــــْ

ومــا عـــاش مـــن مــالــه قــد غــشــاهْ
ومـــا المــال إلا كــأذْن الأصــمــــــــــــْ

وصـــن ســَـــوْءةً عـــن جـــمــيــع الـــســداةْ
تـــعـــش فــي رحـــاب الــعــلــى والــعــلــمْ

فـــمــا غــارقٍ مــات يُـــســمــع نـــداهـــــْ
ومـــا فــاحــشٍ بـــان مــنــه الــكــرمــــــْ

يـــســلــي الــهــوى قـــد هــوى بــهــواهـــْ
وحـــيـــن الـــتـــصــافــي يــكــون عــجــمْ

ألا لـــيـــت عــهــدٌ يـــعـــود ســـنــاهــــــْ
مـــضــى قــد رأيــنــاه حـــبــر الــقــلــمْ
